4. 핵심 알고리즘 분석

검색 및 필터링 알고리즘

// DatabaseManager::getFilteredProperties() - 핵심 검색 로직 QList<Property> DatabaseManager::getFilteredProperties(const SearchFilters& filters) { QList<Property> properties; if (!connectToDatabase()) { return properties; } QString queryStr = "SELECT * FROM properties WHERE 1=1"; QVariantList bindValues; // 검색어 필터 if (!filters.searchTerm.isEmpty()) { queryStr += " AND (address LIKE ? OR details LIKE ? OR description LIKE ?)"; QString searchPattern = QString("%%1%").arg(filters.searchTerm); bindValues << searchPattern << searchPattern << searchPattern; } // 가격 필터 if (filters.minPrice > 0) { queryStr += " AND price >= ?"; bindValues << filters.minPrice; } if (filters.maxPrice < 10000000) { queryStr += " AND price <= ?"; bindValues << filters.maxPrice; } // 침실 수 필터 if (filters.minBeds > 0) { queryStr += " AND beds >= ?"; bindValues << filters.minBeds; } if (filters.maxBeds < 10) { queryStr += " AND beds <= ?"; bindValues << filters.maxBeds; } // 정렬 옵션 적용 if (filters.sortBy == "price_asc") { queryStr += " ORDER BY price ASC"; } else if (filters.sortBy == "price_desc") { queryStr += " ORDER BY price DESC"; } else if (filters.sortBy == "date_desc") { queryStr += " ORDER BY created_at DESC"; } else if (filters.sortBy == "trust_desc") { queryStr += " ORDER BY trust_score DESC"; } // 쿼리 실행 QSqlQuery query(m_database); query.prepare(queryStr); for (const QVariant& value : bindValues) { query.addBindValue(value); } if (query.exec()) { while (query.next()) { properties.append(queryToProperty(query)); } } else { qDebug() << "필터 쿼리 실행 실패:" << query.lastError().text(); } return properties; }

즐겨찾기 관리 알고리즘

// 즐겨찾기 토글 기능 bool DatabaseManager::toggleFavorite(int propertyId) { if (isFavorite(propertyId)) { return removeFromFavorites(propertyId); } else { return addToFavorites(propertyId); } } // 즐겨찾기 추가 bool DatabaseManager::addToFavorites(int propertyId) { if (!connectToDatabase()) { return false; } QSqlQuery query(m_database); query.prepare("INSERT INTO favorites (property_id) VALUES (?)"); query.addBindValue(propertyId); if (!query.exec()) { qDebug() << "즐겨찾기 추가 실패:" << query.lastError().text(); return false; } // properties 테이블의 is_favorite도 업데이트 QSqlQuery updateQuery(m_database); updateQuery.prepare("UPDATE properties SET is_favorite = TRUE WHERE id = ?"); updateQuery.addBindValue(propertyId); updateQuery.exec(); return true; }

데이터 변환 알고리즘

// SQL 쿼리 결과를 Property 객체로 변환 Property DatabaseManager::queryToProperty(const QSqlQuery& query) { Property property; property.id = query.value("id").toInt(); property.imageUrl = query.value("image_url").toString(); // 가격 포맷팅 (한국 원화) double price = query.value("price").toDouble(); if (price >= 10000) { property.price = QString("₩%1억").arg(price / 100000000, 0, 'f', 1); } else { property.price = QString("₩%1").arg(price, 0, 'f', 0); } property.address = query.value("address").toString(); property.details = query.value("details").toString(); property.trustScore = query.value("trust_score").toInt(); property.beds = query.value("beds").toInt(); property.baths = query.value("baths").toInt(); property.sqft = query.value("sqft").toString(); property.propertyType = query.value("property_type").toString(); property.status = query.value("status").toString(); property.yearBuilt = query.value("year_built").toString(); property.lotSize = query.value("lot_size").toString(); property.parking = query.value("parking").toString(); property.heating = query.value("heating").toString(); property.cooling = query.value("cooling").toString(); property.description = query.value("description").toString(); property.latitude = query.value("latitude").toDouble(); property.longitude = query.value("longitude").toDouble(); property.isFavorite = query.value("is_favorite").toBool(); property.createdAt = query.value("created_at").toDateTime(); property.updatedAt = query.value("updated_at").toDateTime(); // 편의시설 파싱 QString amenitiesStr = query.value("amenities").toString(); if (!amenitiesStr.isEmpty()) { property.amenities = amenitiesStr.split(',', Qt::SkipEmptyParts); } // 이미지 URLs 파싱 QString imageUrlsStr = query.value("image_urls").toString(); if (!imageUrlsStr.isEmpty()) { property.imageUrls = imageUrlsStr.split(',', Qt::SkipEmptyParts); } return property; }
