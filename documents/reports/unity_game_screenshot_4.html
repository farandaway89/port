67 void OnGUI()
68 {
69 　　// 4K 해상도 지원을 위한 동적 스케일링

71 　　scaleFactor = Mathf.Clamp(scaleFactor, 1f, 3f);
72
73 　　Matrix4x4 oldMatrix = GUI.matrix;
74 　　GUI.matrix = Matrix4x4.Scale(new Vector3(scaleFactor, scaleFactor, 1f));
75
76 　　float width = 600f;
77 　　float height = 25f;
78 　　float x = 10f;
79
80 　　GUI.Label(new Rect(x, 10, width, height), "🎯 Monument Valley 스파르타 전설");
81 　　GUI.Label(new Rect(x, 40, width, height), "🖱️ WASD: 이동, Space: 점프");
82 　　GUI.Label(new Rect(x, 70, width, height), "⚔️ 마우스 클릭: 공격");
83
84 　　GUI.matrix = oldMatrix; // 매트릭스 복원
85 }
