CREATE TABLE properties ( id INT AUTO_INCREMENT PRIMARY KEY, image_url VARCHAR(500), price DECIMAL(15,2), address VARCHAR(500), details TEXT, trust_score INT DEFAULT 0, beds INT DEFAULT 0, baths INT DEFAULT 0, sqft VARCHAR(50), property_type VARCHAR(100) DEFAULT '아파트', status VARCHAR(50) DEFAULT 'For Sale', year_built VARCHAR(10), lot_size VARCHAR(50), parking VARCHAR(100), heating VARCHAR(100), cooling VARCHAR(100), description TEXT, amenities TEXT, image_urls TEXT, latitude DECIMAL(10,8), longitude DECIMAL(11,8), is_favorite BOOLEAN DEFAULT FALSE, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP );
Favorites 테이블: 사용자별 즐겨찾기 관리
CREATE TABLE favorites ( id INT AUTO_INCREMENT PRIMARY KEY, property_id INT, user_id VARCHAR(100) DEFAULT 'default', created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY (property_id) REFERENCES properties(id) ON DELETE CASCADE );

검색 및 필터링 알고리즘

// DatabaseManager::getFilteredProperties() - 핵심 검색 로직 QList DatabaseManager::getFilteredProperties(const SearchFilters& filters) { QString queryStr = "SELECT * FROM properties WHERE 1=1"; QVariantList bindValues; // 검색어 필터 if (!filters.searchTerm.isEmpty()) { queryStr += " AND (address LIKE ? OR details LIKE ? OR description LIKE ?)"; QString searchPattern = QString("%%1%").arg(filters.searchTerm); bindValues << searchPattern << searchPattern << searchPattern; } // 가격 필터 if (filters.minPrice > 0) { queryStr += " AND price >= ?"; bindValues << filters.minPrice; } if (filters.maxPrice < 10000000) { queryStr += " AND price <= ?"; bindValues << filters.maxPrice; } // 정렬 옵션 적용 if (filters.sortBy == "price_asc") { queryStr += " ORDER BY price ASC"; } else if (filters.sortBy == "price_desc") { queryStr += " ORDER BY price DESC"; } else if (filters.sortBy == "date_desc") { queryStr += " ORDER BY created_at DESC"; } else if (filters.sortBy == "trust_desc") { queryStr += " ORDER BY trust_score DESC"; } // 쿼리 실행 및 결과 반환 QSqlQuery query(m_database); query.prepare(queryStr); for (const QVariant& value : bindValues) { query.addBindValue(value); } QList properties; if (query.exec()) { while (query.next()) { properties.append(queryToProperty(query)); } } return properties; }

즐겨찾기 관리 시스템

// 즐겨찾기 토글 기능 bool DatabaseManager::toggleFavorite(int propertyId) { if (isFavorite(propertyId)) { return removeFromFavorites(propertyId); } else { return addToFavorites(propertyId); } } // 즐겨찾기 추가 bool DatabaseManager::addToFavorites(int propertyId) { QSqlQuery query(m_database); query.prepare("INSERT INTO favorites (property_id) VALUES (?)"); query.addBindValue(propertyId); if (!query.exec()) { return false; } // properties 테이블의 is_favorite도 업데이트 QSqlQuery updateQuery(m_database); updateQuery.prepare("UPDATE properties SET is_favorite = TRUE WHERE id = ?"); updateQuery.addBindValue(propertyId); updateQuery.exec(); return true; }

9. 개발 과정 및 학습 성과

기술적 도전과 해결

Qt6 프레임워크 마스터링

도전: Qt6의 새로운 아키텍처와 위젯 시스템을 익히는 것이 복잡했습니다. 특히 시그널/슬롯 시스템과 메모리 관리, 위젯 계층 구조를 이해하는 데 시간이 걸렸습니다.

해결: Qt 공식 문서와 예제 코드를 꼼꼼히 분석하고, 작은 프로토타입부터 시작하여 점진적으로 기능을 확장해 나가는 방식으로 학습했습니다. 특히 ModernPropertyCard 클래스에서 Qt의 스타일시트 시스템을 활용한 현대적인 UI 구현에 집중했습니다.

데이터베이스 연동 최적화

도전: MySQL과 Qt의 QSqlDatabase를 연동하여 대용량 부동산 데이터를 효율적으로 처리하는 것이 까다로웠습니다. 특히 복잡한 검색 쿼리와 인덱스 최적화가 필요했습니다.

해결: DatabaseManager 클래스를 싱글톤 패턴으로 구현하여 데이터베이스 연결을 효율적으로 관리하고, prepared statement를 사용하여 SQL injection을 방지했습니다. 또한 적절한 인덱스를 생성하여 검색 성능을 최적화했습니다.

핵심 학습 포인트

  • 현대적인 GUI 개발: Qt6의 강력한 위젯 시스템과 스타일시트를 활용하여 사용자 친화적인 인터페이스 구현

  • 메모리 관리: Qt의 부모-자식 관계를 통한 자동 메모리 관리

가장 인상깊었던 부분은 Qt의 시그널/슬롯 시스템을 활용한 모듈 간 통신과 데이터베이스 연동을 통한 효율적인 데이터 관리였습니다. 이러한 기술들을 통해 복잡한 기능을 직관적이고 안정적으로 구현할 수 있다는 것을 깨달았습니다.

앞으로도 이번 프로젝트에서 습득한 기술과 경험을 바탕으로 더욱 혁신적이고 사용자 친화적인 소프트웨어 개발에 지속적으로 기여하고 싶습니다. 특히 부동산과 같은 실생활 분야의 디지털 혁신에 기여할 수 있어 큰 보람을 느낍니다.
